**代码实现**:

# Z-Score HIBOR交易策略
def calculate_zscore(hibor, window=20):
    sma = hibor.rolling(window).mean()
    std = hibor.rolling(window).std()
    return (hibor - sma) / std

def trade_zscore_strategy(current_hibor, zscore_20):
    if zscore_20 < -1.5:
        return 'BUY'  # HIBOR低估
    elif zscore_20 > 1.5:
